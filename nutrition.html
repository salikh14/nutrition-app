<html lang="en">

<head>

    <title>Nutrition Info</title>

    <style>

        body {

            font-family: Arial, sans-serif;

            margin: 20px;

        }

        #results {

            margin-top: 20px;

        }

    </style>

</head>

<body>

    <h1>Nutrition Information</h1>

    <div id="results">Waiting for input...</div>


    <script>

        async function fetchNutrition(food) {

            const appId = "99f5cdd7"; 

            const appKey =  "68681a8757b7f873ff5ee1f82e2d4409 —"; 

            const url = "https://trackapi.nutritionix.com/v2/natural/nutrients";


            try {

                const response = await fetch(url, {

                    method: "POST",

                    headers: {

                        "x-app-id": appId,

                        "x-app-key": appKey,

                        "Content-Type": "application/json"

                    },

                    body: JSON.stringify({ query: food })

                });


                const data = await response.json();

                document.getElementById("results").innerHTML = `

                    <p>Food: ${food}</p>

                    <p>Calories: ${data.foods[0].nf_calories}</p>

                    <p>Carbs: ${data.foods[0].nf_total_carbohydrate}</p>

                    <p>Fats: ${data.foods[0].nf_total_fat}</p>

                `;

            } catch (error) {

                document.getElementById("results").innerHTML = `<p>Error fetching data.</p>`;

                console.error(error);

            }

        }


        // Listen for input from MIT App Inventor WebViewer

        setInterval(() => {

            const food = window.AppInventor.getWebViewString();

            if (food) {

                fetchNutrition(food);

            }

        }, 1000);

    </script>

</body>

</html>
